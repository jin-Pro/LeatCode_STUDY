int* twoSum(int* nums, int numsSize, int target, int* returnSize) {
    int* ptr = (int*)malloc(sizeof(int) *2);
    int i = 0;
    int j= 0;
    *returnSize=2;
    for (i=0;i<numsSize-1;i++) {
        for(j=i+1;j<numsSize;j++)         
        {
           if((nums[i]+nums[j])==target) 
           {
               
            ptr[0] = i;
            ptr[1] = j;
            break;
        
           }
        }    

    }
    return ptr;
}
