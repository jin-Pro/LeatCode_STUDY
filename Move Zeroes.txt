class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int m=0;
        if(nums.size()!=1){
        for(int i=0;i<nums.size();i++){
            if(nums[i]!=0){
                if(i==m){
                    m++;
                }
                else{
                nums[m]=nums[i];
                nums[i]=0;
                m++;
                }
            }
            }
        }
    }
};
